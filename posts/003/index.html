    Take #3 #

    VBA code snippets

    Sub wordDBFill()

Do
If ActiveCell.value <> "" Then
    ActiveCell.Offset(1, 0).Activate
Else
    ActiveCell.value = ActiveCell.End(xlUp).value
    ActiveCell.Offset(1, 0).Activate
End If
Loop Until ActiveCell.Offset(0, 1).value = ""

End Sub
